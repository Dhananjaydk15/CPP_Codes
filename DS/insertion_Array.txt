void display(int arr[], int n)
{
    for (int i = 0; i < n; i++)
    {
        cout << "\n* Print " << i + i << " Element of Array : " << arr[i] << endl;
    };
    
}

int indInsertion(int arr[], int size, int element, int capacity, int index)
{
    int temp = size ;
    if (size >= capacity)
    {
        return -1;
    };
    for (int i = size - 1 ; i >= index ; i--)
    {
        arr[i + 1] = arr[i];
    };
    arr[index] = element;
    return temp;
}
